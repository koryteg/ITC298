				function gone(){
					$(this).fadeOut();

				}

				function style1(){
					if ($("body").hasClass("style1")){
						$("body").removeClass("style1");
					}
					$("body").addClass("style2");
				};

				function style2(){
					if ($("body").hasClass("style2")){
						$("body").removeClass("style2");
					}
					$("body").addClass("style1");
				}
